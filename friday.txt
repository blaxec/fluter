        // --- АУДИОСИСТЕМА (УЛЬТРА-СОВМЕСТИМАЯ) ---
        let audioCtx = null;
        let speechSynth = window.speechSynthesis;

        // Прямая разблокировка по действию пользователя
        function forceUnlockAudio() {
            try {
                if (!audioCtx) {
                    audioCtx = new (window.AudioContext || window.webkitAudioContext)();
                }
                if (audioCtx && audioCtx.state === 'suspended') {
                    audioCtx.resume();
                }
            } catch (e) {
                console.warn("Ошибка разблокировки Web Audio API", e);
            }
        }

        // Генерация чистого пика
        function playTone(freq, duration, type = 'sine') {
            if (state.isMuted) return;
            try {
                forceUnlockAudio();
                if (!audioCtx) return;

                const osc = audioCtx.createOscillator();
                const gain = audioCtx.createGain();
                
                osc.connect(gain);
                gain.connect(audioCtx.destination);
                
                osc.type = type;
                osc.frequency.setValueAtTime(freq, audioCtx.currentTime);
                
                // Исключаем щелчок при включении и выключении
                gain.gain.setValueAtTime(0, audioCtx.currentTime);
                gain.gain.linearRampToValueAtTime(state.volume * 0.5, audioCtx.currentTime + 0.01);
                gain.gain.exponentialRampToValueAtTime(0.0001, audioCtx.currentTime + duration);
                
                osc.start(audioCtx.currentTime);
                osc.stop(audioCtx.currentTime + duration);
            } catch (e) {
                console.warn("Ошибка генерации звука:", e);
            }
        }

        // Синтез речи (с полной очисткой очереди и надежным перезапуском)
        function speak(text) {
            if (state.isMuted) return;
            
            // Если в браузере заблокирован TTS, дублируем команду писком разной тональности
            playTone(660, 0.15, 'triangle');

            if (!speechSynth) return;
            try {
                speechSynth.cancel(); // Очищаем всё перед новой фразой
                
                const utterance = new SpeechSynthesisUtterance(text);
                utterance.lang = 'ru-RU';
                utterance.volume = state.volume;
                utterance.rate = 1.1; // Чуть быстрее для динамичности тренировки
                
                // Принудительное назначение русского голоса
                const voices = speechSynth.getVoices();
                const ruVoice = voices.find(v => v.lang.includes('RU') || v.lang.includes('ru'));
                if (ruVoice) {
                    utterance.voice = ruVoice;
                }

                speechSynth.speak(utterance);
            } catch (e) {
                console.warn("Синтез речи заблокирован системой или недоступен:", e);
            }
        }

        function playTransitionBeep() {

        }

        function updateCircleColor(color) {
            const glowColors = {
                blue: "rgba(59, 130, 246, 0.15)",
                red: "rgba(239, 68, 68, 0.15)",
                orange: "rgba(249, 115, 22, 0.15)",
                green: "rgba(16, 185, 129, 0.15)",
                purple: "rgba(168, 85, 247, 0.15)"
            };
            progressCircle.setAttribute('stroke', `url(#${color}Gradient)`);
            statusBgGlow.style.backgroundColor = glowColors[color] || "rgba(59, 130, 246, 0.15)";
        }

        function updateProgress(remaining, total) {
            if (total <= 0) {
                progressCircle.style.strokeDashoffset = 0;
                return;
            }
            const radius = 42;
            const circumference = 2 * Math.PI * radius; 
            const offset = circumference - (remaining / total) * circumference;
            progressCircle.style.strokeDashoffset = offset;
        }

            const activeStepId = getCurrentExerciseConfig().id;

                if (id === activeStepId) {

                    el.classList.remove('opacity-55');
                    if (icon) icon.classList.add('bg-indigo-500', 'border-indigo-400', 'text-white');

                    el.classList.remove('bg-indigo-950/40', 'border-indigo-500/30');
                    el.classList.add('opacity-55');
                    if (icon) icon.classList.remove('bg-indigo-500', 'border-indigo-400', 'text-white');
                }
            });
        }

        function recordStatEntry(label, desc) {
            const container = document.getElementById('statsContainer');
            const placeholder = document.getElementById('statsPlaceholder');
            placeholder.classList.add('hidden');
            container.classList.remove('hidden');

        function startTimer() {
            if (state.isActive) return;
            
            forceUnlockAudio();

            state.isActive = true;
            document.getElementById('btnStart').classList.add('hidden');
            document.getElementById('btnPause').classList.remove('hidden');

            const currentPhase = getCurrentExerciseConfig().steps[state.phaseIndex];
            if (currentPhase.voice) speak(currentPhase.voice);

            state.timerId = setInterval(() => {
                const exConfig = getCurrentExerciseConfig();
                const phase = exConfig.steps[state.phaseIndex];

                if (phase.isManual) {
                    clearInterval(state.timerId);
                    return;
                }

                if (state.timeRemaining > 0) {
                    state.timeRemaining--;

                    // Предупреждающие звуки только за 3 секунды до конца фазы
                    if (state.timeRemaining > 0 && state.timeRemaining <= 3) {
                        playPreBeep();
                    }
                    renderTimer();
                } else {
                    playTransitionBeep();
                    nextPhase();
                }
            }, 1000);
        }

        function pauseTimer() {
            state.isActive = false;
            clearInterval(state.timerId);
            document.getElementById('btnStart').classList.remove('hidden');
            document.getElementById('btnPause').classList.add('hidden');
        }

        function resetTimer() {
            pauseTimer();
            state.currentCircuit = 1;
            state.exerciseIndex = 0;
            state.setIndex = 1;
            state.phaseIndex = 0;
            state.activeRestOnComplete = null;
            
            const exConfig = getCurrentExerciseConfig();
            state.timeRemaining = exConfig.steps[0].duration;
            state.totalDuration = state.timeRemaining;

            state.stats = [];
            document.getElementById('statsContainer').innerHTML = '';
            document.getElementById('statsPlaceholder').classList.remove('hidden');
            document.getElementById('statsContainer').classList.add('hidden');

            renderTimer();
        }

        function nextPhase() {
            const exConfig = getCurrentExerciseConfig();
            
            if (state.exerciseIndex === 0 && state.phaseIndex === exConfig.steps.length - 1) {
                recordStatEntry(`Негативные подтягивания (Подход ${state.setIndex})`, `Выполнено за 15с`);
            }

            state.phaseIndex++;

            if (state.phaseIndex < exConfig.steps.length) {
                const nextStep = exConfig.steps[state.phaseIndex];
                state.timeRemaining = nextStep.duration;
                state.totalDuration = nextStep.duration;
                
                if (nextStep.voice) speak(nextStep.voice);
                renderTimer();
            } else {
                goToNextExercise();
            }
        }

        function goToNextExercise() {
            if (state.exerciseIndex === 0) {
                if (state.setIndex < 2) {
                    state.setIndex = 2;
                    state.phaseIndex = 0;
                    
                    startRestPhase(config.restNeg, "Отдых между подходами", () => {
                        const nextEx = getCurrentExerciseConfig();
                        state.timeRemaining = nextEx.steps[0].duration;
                        state.totalDuration = state.timeRemaining;
                        if (nextEx.steps[0].voice) speak(nextEx.steps[0].voice);
                        renderTimer();
                    });
                } else {
                    // Пауза-подготовка перед переходом к жиму одной руки (15 секунд)
                    startRestPhase(15, "Подготовка к жиму", () => {
                        state.exerciseIndex = 1;
                        state.setIndex = 1;
                        state.phaseIndex = 0;
                        initNextExerciseTime();
                    }, "Приготовьте гантели", "blue", "Приготовьтесь к жиму гантелей, пятнадцать секунд");
                }
            } else if (state.exerciseIndex === 1) {
                if (state.setIndex === 1) {
                    state.setIndex = 2;
                    state.phaseIndex = 0;
                    
                    startRestPhase(5, "Смена руки", () => {
                        const nextEx = getCurrentExerciseConfig();
                        state.timeRemaining = nextEx.steps[0].duration;
                        state.totalDuration = state.timeRemaining;
                        if (nextEx.steps[0].voice) speak(nextEx.steps[0].voice);
                        renderTimer();
                    });
                } else {
                    recordStatEntry(`Жим гантелей (Обе руки)`, `${config.pressReps}x повторов`);
                    
                    // Пауза-подготовка перед переходом к турнику (10 секунд)
                    startRestPhase(10, "Переход к турнику", () => {
                        state.exerciseIndex = 2;
                        state.setIndex = 1;
                        state.phaseIndex = 0;
                        initNextExerciseTime();
                    }, "Займите позицию на турнике", "blue", "Переход к турнику, десять секунд");
                }
            } else {
                const userReps = parseInt(document.getElementById('txtInteractiveReps').textContent) || config.revReps;
                recordStatEntry(`Обратные подтягивания`, `${userReps} повторов`);
                
                state.stats.push({
                    circuit: state.currentCircuit,
                    revReps: userReps
                });

                if (state.currentCircuit < config.circuitsCount) {
                    state.currentCircuit++;
                    state.exerciseIndex = 0;
                    state.setIndex = 1;
                    state.phaseIndex = 0;

                    startRestPhase(config.circuitRest, "Отдых между кругами", () => {
                        initNextExerciseTime();
                    });
                } else {
                    finishWorkout();
                }
            }
        }

        function initNextExerciseTime() {
            const exConfig = getCurrentExerciseConfig();
            state.timeRemaining = exConfig.steps[0].duration;
            state.totalDuration = state.timeRemaining;
            if (exConfig.steps[0].voice) speak(exConfig.steps[0].voice);
            renderTimer();
        }

        function startRestPhase(duration, label, onComplete, subState = "Время отдыха", color = "purple", customVoice = null) {
            pauseTimer();
            state.isActive = true;
            document.getElementById('btnStart').classList.add('hidden');
            document.getElementById('btnPause').classList.remove('hidden');

            state.timeRemaining = duration;
            state.totalDuration = duration;
            state.activeRestOnComplete = onComplete; // Сохраняем коллбэк для возможности пропуска
            
            document.getElementById('txtExerciseName').textContent = label;
            document.getElementById('txtSubState').textContent = subState;
            updateCircleColor(color);
            speak(customVoice || `Отдых ${duration} секунд`);

            renderTimer();

            state.timerId = setInterval(() => {
                if (state.timeRemaining > 0) {
                    state.timeRemaining--;
                    if (state.timeRemaining > 0 && state.timeRemaining <= 3) playPreBeep();
                    renderTimer();
                } else {
                    playTransitionBeep();
                    clearInterval(state.timerId);
                    state.isActive = false;
                    state.activeRestOnComplete = null;
                    onComplete();
                    startTimer(); 
                }
            }, 1000);
        }

        // Функция пропуска текущего шага
        function skipCurrentPhase() {
            forceUnlockAudio();
            playTransitionBeep();
            
            if (state.timerId) {
                clearInterval(state.timerId);
            }
            state.isActive = false;

            // Если пропускаем фазу отдыха / перехода
            if (state.activeRestOnComplete) {
                const callback = state.activeRestOnComplete;
                state.activeRestOnComplete = null;
                callback();
                startTimer();
            } else {
                // Если пропускаем обычную фазу упражнения
                const exConfig = getCurrentExerciseConfig();
                const phase = exConfig.steps[state.phaseIndex];
                
                if (phase && phase.isManual) {
                    // Если это обратные подтягивания (ручной ввод), сразу переходим к следующему шагу
                    goToNextExercise();
                    startTimer();
                } else {
                    nextPhase();
                    startTimer();
                }
            }
        }

        }

            config.prepTime = parseInt(document.getElementById('cfg-prep').value);

        // --- ИНТЕГРАЦИЯ С GEMINI API ---
        async function fetchGeminiWithRetry(prompt, systemInstruction = "", retries = 5, delay = 1000) {
            const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
            const payload = {
                contents: [{ parts: [{ text: prompt }] }]
            };
            if (systemInstruction) {
                payload.systemInstruction = { parts: [{ text: systemInstruction }] };
            }

            for (let i = 0; i < retries; i++) {
                try {
                    const response = await fetch(url, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });
                    if (response.ok) {
                        return await response.json();
                    }
                } catch (e) {
                    if (i === retries - 1) throw e;
                }
                await new Promise(res => setTimeout(res, delay * Math.pow(2, i)));
            }
            throw new Error("Не удалось получить ответ от Gemini API");
        }

        document.getElementById('btnAiConfigure').addEventListener('pointerdown', async () => {
            const userPrompt = document.getElementById('aiConfigInput').value.trim();
            if (!userPrompt) return;

            const btn = document.getElementById('btnAiConfigure');
            const originalText = btn.innerHTML;
            btn.innerHTML = `<span class="animate-spin mr-2">⏳</span> Анализ...`;
            btn.disabled = true;

            const systemPrompt = `Ты профессиональный спортивный тренер. Разбери запрос пользователя и верни настройки тренировки в формате JSON.
Твой ответ должен содержать только валидный JSON-объект без разметки markdown со следующей схемой:
{
  "prepTime": число (3-15),
  "circuitsCount": число (1-5),
  "negTime1": число (2-15),
  "negTime2": число (2-15),
  "negTime3": число (2-15),
  "restNeg": число (5-60),
  "pressReps": число (5-50),
  "pressHold": число (1-5),
  "pressDrop": число (1-5),
  "revReps": число (2-20),
  "circuitRest": число (10-180),
  "explanation": "Короткое пояснение внесенных изменений"
}`;

            try {
                const response = await fetchGeminiWithRetry(userPrompt, systemPrompt);
                const text = response.candidates?.[0]?.content?.parts?.[0]?.text;
                const jsonMatch = text.match(/\{[\s\S]*\}/);
                if (jsonMatch) {
                    const data = JSON.parse(jsonMatch[0]);
                    
                    config.prepTime = data.prepTime || config.prepTime;
                    config.circuitsCount = data.circuitsCount || config.circuitsCount;
                    config.negTime1 = data.negTime1 || config.negTime1;
                    config.negTime2 = data.negTime2 || config.negTime2;
                    config.negTime3 = data.negTime3 || config.negTime3;
                    config.restNeg = data.restNeg || config.restNeg;
                    config.pressReps = data.pressReps || config.pressReps;
                    config.pressHold = data.pressHold || config.pressHold;
                    config.pressDrop = data.pressDrop || config.pressDrop;
                    config.revReps = data.revReps || config.revReps;
                    config.circuitRest = data.circuitRest || config.circuitRest;

                    resetTimer();

                    document.getElementById('aiResponseText').textContent = `⚙️ Настройки применены!\n\n${data.explanation}`;
                    document.getElementById('aiResponseBox').classList.remove('hidden');
                }
            } catch (e) {
                console.error(e);
                document.getElementById('aiResponseText').textContent = "Не удалось обновить параметры ИИ.";
                document.getElementById('aiResponseBox').classList.remove('hidden');
            } finally {
                btn.innerHTML = originalText;
                btn.disabled = false;
            }
        });
